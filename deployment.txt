1) Create the deployment 
vim dp.yml
apiVersion: apps/v1
kind: Deployment

metadata:
  name: dep1

spec:
  replicas: 3
  selector:
    matchLabels:
      red: green
  template:
    metadata:
      labels:
        red: green
    spec:
      containers:
      - name: nginx
        image: nginx:1.14.2
